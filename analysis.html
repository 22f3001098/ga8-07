# Employee Performance Analysis
# Verification Email: 22f3001098@ds.study.iitm.ac.in

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Sample employee data (50 employees for realistic distribution)
data = {
    'Employee_ID': list(range(1, 51)),
    'Name': [f'Employee_{i}' for i in range(1, 51)],
    'Department': ['Marketing']*13 + ['Sales']*12 + ['IT']*10 + ['HR']*8 + ['Finance']*7,
    'Region': ['North', 'South', 'East', 'West'] * 12 + ['North', 'South'],
    'Performance_Score': [85, 92, 78, 88, 95, 82, 90, 87, 79, 91,
                          84, 89, 86, 93, 80, 88, 85, 91, 87, 83,
                          94, 77, 89, 92, 81, 86, 90, 84, 88, 93,
                          79, 85, 91, 87, 82, 89, 94, 76, 88, 92,
                          83, 87, 90, 85, 91, 84, 88, 93, 86, 89]
}

# Create DataFrame
df = pd.DataFrame(data)

print("=" * 60)
print("EMPLOYEE PERFORMANCE ANALYSIS")
print("Verification Email: 22f3001098@ds.study.iitm.ac.in")
print("=" * 60)

# Calculate frequency count for Marketing department
marketing_count = df[df['Department'] == 'Marketing'].shape[0]
print(f"\nFrequency count for Marketing department: {marketing_count}")

# Calculate frequency for all departments
dept_frequency = df['Department'].value_counts()
print("\nDepartment Frequency Distribution:")
print(dept_frequency)

# Create histogram showing distribution of departments
plt.figure(figsize=(12, 7))
sns.set_style("whitegrid")

# Create bar plot (histogram for categorical data)
dept_counts = df['Department'].value_counts().sort_values(ascending=False)
colors = ['#667eea', '#764ba2', '#f093fb', '#4facfe', '#43e97b']
bars = plt.bar(dept_counts.index, dept_counts.values, color=colors)

# Highlight Marketing department
bars[0].set_color('#e74c3c')

plt.xlabel('Department', fontsize=14, fontweight='bold')
plt.ylabel('Number of Employees', fontsize=14, fontweight='bold')
plt.title('Employee Distribution by Department\nAnalysis by: 22f3001098@ds.study.iitm.ac.in', 
          fontsize=16, fontweight='bold', pad=20)
plt.xticks(rotation=45, ha='right')
plt.grid(axis='y', alpha=0.3)

# Add value labels on bars
for i, (dept, count) in enumerate(dept_counts.items()):
    plt.text(i, count + 0.3, str(count), ha='center', va='bottom', 
             fontweight='bold', fontsize=12)

plt.tight_layout()

# Save the plot
plt.savefig('department_distribution.png', dpi=300, bbox_inches='tight')
print("\nVisualization saved as 'department_distribution.png'")

plt.show()

# Additional Analysis
print("\n" + "=" * 60)
print("SUMMARY STATISTICS")
print("=" * 60)
print(f"Total Employees: {len(df)}")
print(f"Number of Departments: {df['Department'].nunique()}")
print(f"Number of Regions: {df['Region'].nunique()}")
print(f"\nMarketing Department Count: {marketing_count}")
print(f"Marketing Percentage: {(marketing_count/len(df)*100):.1f}%")
print("=" * 60)
